public class midterm {

    static public int searchcount(int x[], int target)
    {
        int count = 0;
        for (int i = 0; i < x.length; i++)
        {
            if (x[i] == target) count ++;
        }
        return count;
    }


    
public static void main(String args[])
        {
            System.out.println(searchcount(new int[]{1,3,2,3}, 3));
            System.out.println(searchcount(new int[]{1,3,2,3}, 4));
        }
    }

int binarysearch( int x[], int low, int high)
{
	while (low <= high)
	{
		int m = (low + high)/2;
		if (x[m] > target)
			high = m - 1;
		else if (x[m] < target)
			low = m + 1;
		return m; 
	}
	return -m;
}

int maxima(int [] x)
{
	int max = 0;
	for (int i = 1; i < x.length; i++)	
	{
		if (x[i] > x[max])
		max = i;
	}
	return max;
}
